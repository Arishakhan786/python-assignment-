THE LAST TOPIC LINE OF REGRESSION

In[31]:
import pandas as pd
import matplotlib.pyplot as plt
import numpy as np
from sklearn.linear_model import LinearRegression
data = pd.read_csv('Youth_Tobacco_Survey__YTS__Data.csv')
smoking_data = data[data['TopicDesc'] == 'Cigarette Use (Youth)']
grouped_data = smoking_data.groupby('YEAR')['Data_Value'].mean().reset_index()
X = grouped_data['YEAR'].values.reshape(-1, 1)
y = grouped_data['Data_Value'].values
model = LinearRegression()
model.fit(X, y)
y_pred = model.predict(X)
plt.figure(figsize=(10, 6))
plt.scatter(X, y, color='blue', label='Actual Data')
plt.plot(X, y_pred, color='red', linewidth=2, label='Linear Regression Line')
plt.xlabel('Year')
plt.ylabel('Smoking Prevalence')
plt.title('Linear Regression of Smoking Prevalence Over the Years')
plt.legend()
plt.grid(True)
plt.tight_layout()
plt.show()
